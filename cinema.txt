Procedure Cinema 

begin 
		
		Numeric nTime 
		
		Display 
		
					cTitleChose
			
					cInfor[nCinema]
		
					cPrice[nCinema]
					
					cTeater[nCinema]
		
		Time : 
		0. 09.00 AM
		1. 11.00 AM
		2. 13.00 PM
		3. 15.00 PM		
		"
		
		Accept nTime
		
		Switch (nTime)
		begin
				case 0 :
						nLimitSeat = nSeatTeater[nCinema] * 1
						nStartSeat = nLimitSeat - nSeatTeater[nCinema]
						cTime2 = '09.00 AM' 
						call Proce seat
				Break
				
				Case 1 :
						nLimitSeat = nSeatTeater[nCinema] * 2
						nStartSeat = nLimitSeat - nSeatTeater[nCinema]
						cTime 2 = '11.00 AM'
						call Proce seat
				Break
				Case 2 :
						nLimitSeat = nSeatTeater[nCinema] * 3
						nStartSeat = nLimitSeat - nSeatTeater[nCinema]
						cTime 2 = '13.00 AM'
						call Proce seat
				Break
				Case 3 :
						nLimitSeat = nSeatTeater[nCinema] * 4
						nStartSeat = nLimitSeat - nSeatTeater[nCinema]
						cTime 2 = '15.00 AM'
						call Proce seat
				Break
				Default :
						Display 'The code is not Available'
						call Proce Cinema
				Break
		end
end
